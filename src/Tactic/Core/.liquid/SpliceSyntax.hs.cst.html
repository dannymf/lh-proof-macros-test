 50: freshName :: String -> Splice Name
 51: freshName str = do
 52:   env <- get
 53:   let i = var_i env
 54:   let name = mkName $ str ++ "_" ++ show i
 55:   put $ env {var_i = i + 1}
 56:   pure name

 58: spliceExp :: [Instr] -> Splice PreExp
 59: spliceExp instrs =
 60:   if not (null instrs)
 61:     then case last instrs of
 62:       Trivial -> go instrs
 63:       Auto _ _ -> go instrs
 64:       _ -> go (instrs ++ [Auto {hints = defaultAutoHints, depth = defaultAutoDepth}])
 65:     else go instrs
 66:   where
 67:     go :: [Instr] -> Splice PreExp
 68:     go [] = pure TrivialPreExp
 69:     go (Intro {name} : instrs) = do
 70:       name <- pure $ mkName name
 71:       env <- get
 72:       types <- gets def_argTypes
 73:       def_type <- gets def_type
 74:       i <- gets arg_i
 75:       env <- get
 76:       case types `index` i of
 77:         Just type_ -> do modify $ introArg name type_
 78:         Nothing -> fail $ "Cannot intro " ++ show name ++ " at index  " ++ show i ++ " in def_type " ++ pprint def_type
 79:       Lambda name <$> go instrs
 80:     go (Destruct {name, intros, flags} : instrs) = do
 81:       name <- pure $ mkName name
 82:       env <- get
 83:       mb_type <- get >>= lift . inferType (VarE name)
 84:       case mb_type of
 85:         (Just type_@(ConT dtName)) -> local do
 86:           -- remove destructed target from environment
 87:           when (not $ "remember" `elem` flags) $ do
 88:             modify $ deleteCtx (VarE name)
 89:           -- get datatype info
 90:           dtInfo <- lift $ reifyDatatype dtName
 91:           -- gen matches
 92:           let dtConInfos = datatypeCons dtInfo
 93: 
 94:           let matches :: [Splice (Pat, PreExp)]
 95:               matches =
 96:                 ( \i conInfo -> local do
 97:                     -- collects newly bound variables with types, generates match's pattern
 98:                     (vars, pat) <- getConVarsPat conInfo (indexIntros intros i)
 99:                     -- adds constructor's introduced terms to environment
100:                     modify $ flip (foldl (\env (e, type_) -> insertCtx e type_ env)) (fmap (\(n, t) -> (VarE n, t)) vars)
101:                     -- body
102:                     expr <- go instrs
103:                     -- gen match
104:                     pure (pat, expr)
105:                 )
106:                   `mapWithIndex` dtConInfos
107:           -- generate case
108:           ms <- sequence matches
109:           --
110:           pure $ Case (VarE name) ms
111:         Just type_ -> fail $ "Cannot destruct " ++ show name ++ " of non-datatype type " ++ show type_
112:         Nothing -> go instrs -- skip this instruction, since target not in scope
113:     go (Induct {name, intros, flags} : instrs) = do
114:       name <- pure $ mkName name
115:       env <- get
116:       mb_type <- get >>= \env -> lift $ inferType (VarE name) env
117: 
118:       case mb_type of
119:         Just type_@(ConT dtName) -> local do
120:           -- remove inducted target from environment
121:           when (not $ "remember" `elem` flags) $ do
122:             modify $ deleteCtx (VarE name)
123:           -- get datatype info
124:           dtInfo <- lift $ reifyDatatype dtName
125:           -- gen matches
126:           let dtConInfos = datatypeCons dtInfo
127:           let matches :: [Splice (Pat, PreExp)]
128:               matches =
129:                 ( \i conInfo -> local do
130:                     -- collects newly bound variables with types, generates match's pattern
131:                     (vars, pat) <- getConVarsPat conInfo (indexIntros intros i)
132:                     -- add constructor's variables to `args_rec_ctx` at `name`
133:                     (List.elemIndex name <$> gets def_argNames) >>= \case
134:                       Just arg_i -> do
135:                         modify $ \env -> env {args_rec_ctx = Map.insert arg_i (Map.fromList . fmap (\(n, t) -> (VarE n, t)) $ vars) (args_rec_ctx env)}
136:                       Nothing -> do
137:                         -- TODO: dont need this anymore, since can induct on non-arguments: fail $ "Cannot find argument index of argument " ++ show name
138:                         -- ! however, this could potentially cause problems where it's possible to auto your way into infinite recursion, right?
139:                         return ()
140:                     -- adds constructor's introduced terms to environment
141:                     modify $ flip (foldl (flip (uncurry insertCtx))) (fmap (\(n, t) -> (VarE n, t)) $ vars)
142:                     --
143:                     expr <- go instrs
144:                     -- gen match
145:                     pure $ (pat, expr)
146:                 )
147:                   `mapWithIndex` dtConInfos
148:           -- generate case
149:           ms <- sequence matches
150:           --
151:           pure $ Case (VarE name) ms
152:         Just type_ -> fail $ "Cannot induct " ++ show name ++ " of non-datatype type " ++ show type_
153:         Nothing -> go instrs -- skip this instruction, since target not in scope
154:     go (Assert {exp, requires} : instrs) = do
155:       env <- get
156:       inScope <- lift $ all isJust <$> mapM (\x -> inferType (nameToExp $ mkName x) env) requires
157:       if inScope
158:         then If exp <$> go instrs <*> pure TrivialPreExp
159:         else go instrs
160:     go (Dismiss {exp, requires} : instrs) = do
161:       env <- get
162:       inScope <- lift $ all isJust <$> mapM (\x -> inferType (nameToExp $ mkName x) env) requires
163:       if inScope
164:         then flip (If exp) <$> go instrs <*> pure TrivialPreExp
165:         else go instrs
166:     go (Use {exp, requires} : instrs) = do
167:       env <- get
168:       inScope <- lift $ all isJust <$> mapM (\x -> inferType (nameToExp $ mkName x) env) requires
169:       if inScope
170:         then Exp exp <$> go instrs
171:         else go instrs
172:     go (Cond {exp, requires} : instrs) = do
173:       env <- get
174:       inScope <- lift $ all isJust <$> mapM (\x -> inferType (nameToExp $ mkName x) env) requires
175:       if inScope
176:         then do
177:           pe <- go instrs
178:           pure $ If exp pe pe
179:         else go instrs
180:     go (Trivial : instrs) =
181:       go instrs
182:     go (Refine {string, requires} : instrs) = do
183:       -- env <- get
184:       -- refinement <- interpretRefinement string
185:       error "unimplemented: refine"
186:     go (Auto {hints, depth} : instrs) = do
187:       env <- get
188:       -- ctx' <- lift $ Map.fromList <$> mapM (\x -> (x,) <$> inferType x env) hints
189:       let typings =
190:             mapM
191:               ( \x -> do
192:                   mb_type <- inferType x env
193:                   case mb_type of
194:                     Just type_ -> pure [(x, type_)]
195:                     Nothing -> pure []
196:               )
197:               hints
198:       ctx' <- lift $ Map.fromList <$> (concat <$> typings)
199:       proof <- lift [t|Proof|]
200:       es <-
201:         withStateT
202:           (\env -> env {ctx = Map.union ctx' (ctx env)})
203:           $ genNeutrals (Just proof) depth
204:       AutoPreExp es initPruneAutoState <$> go instrs

209: mapWithIndex :: (Int -> a -> b) -> [a] -> [b]
210: mapWithIndex f xs = go 0 xs
211:   where
212:     go i [] = []
213:     go i (x : xs) = f i x : go (i + 1) xs

228: getConVarsPat :: ConstructorInfo -> Maybe [String] -> Splice ([(Name, Type)], Pat)
229: getConVarsPat conInfo mb_intro = do
230:   let conName = constructorName conInfo
231:   let conFields = constructorFields conInfo
232:   names <-
233:     case mb_intro of
234:       Just intro -> pure $ mkName <$> intro
235:       Nothing -> mapM typeToTermName conFields
236:   let vars = zip names conFields
237:   let pat = ConP conName conFields $ VarP <$> names
238:   pure (vars, pat)

240: typeToTermName :: Type -> Splice Name
241: typeToTermName type_ =
242:   case type_ of
243:     ConT name -> case nameBase name of
244:       (c : s) -> freshName (Char.toLower c : s)
245:     _ -> freshName "x"

247: type Goal = Maybe Type

249: type Gas = Int
250: 
251: genNeutrals :: Goal -> Gas -> Splice [Exp]
252: genNeutrals goal 0 = pure []
253: genNeutrals goal gas = do
254:   vars <- Map.toList <$> gets ctx
255:   let f :: [Exp] -> (Exp, Type) -> Splice [Exp]
256:       f es (e, alpha) =
257:         (es <>)
258:           <$> let (_, beta) = flattenType alpha
259:                in if matchesGoal beta goal
260:                     then genNeutrals' e alpha gas
261:                     else pure []
262:   es <- (<>) <$> foldM f [] vars <*> genRecursions goal gas
263:   pure es
264: 
265: genNeutrals' :: Exp -> Type -> Gas -> StateT Environment Q [Exp]
266: genNeutrals' e type_ gas = do
267:   let (alphas, beta) = flattenType type_
268:   es <-
269:     if List.null alphas
270:       then pure [e]
271:       else do
272:         argss <- fanout <$> traverse (\alpha -> genNeutrals (Just alpha) (gas - 1)) alphas
273:         let es = foldl AppE e <$> argss
274:         pure es
275:   -- debugSplice $! "genNeutrals' (" ++ pprint e ++ ") (" ++ pprint type_ ++ ") " ++ show gas ++ " = " ++ show (pprint <$> es)
276:   pure es
277: 
278: -- | generates any expressions directly from context (no applications) that have goal type
279: genAtomsFromCtx :: Ctx -> Type -> Splice [Exp]
280: genAtomsFromCtx ctx type_ = do
281:   let f :: [Exp] -> (Exp, Type) -> Splice [Exp]
282:       f es (e, alpha) =
283:         (es <>) <$> if alpha `compareTypes` type_ then pure [e] else pure []
284:   es <- foldM f [] (Map.toList ctx)
285:   pure es
286: 
287: genRecursions :: Goal -> Int -> Splice [Exp]
288: genRecursions goal gas = do
289:   canRecurse >>= \case
290:     True -> do
291:       r <- VarE <$> gets def_name
292:       rho <- gets def_type
293:       let (alphas, beta) = flattenType rho
294:       if matchesGoal beta goal
295:         then do
296:           if List.null alphas
297:             then fail "impossible: cannot recurse with 0 arguments"
298:             else do
299:               argss <-
300:                 fanout
301:                   <$> traverse
302:                     ( \(arg_i, alpha) -> do
303:                         (Map.lookup arg_i <$> gets args_rec_ctx) >>= \case
304:                           Just rec_ctx -> genAtomsFromCtx rec_ctx alpha -- gen only vars from ctx
305:                           Nothing -> genNeutrals (Just alpha) (gas - 1) -- gen any neutral
306:                     )
307:                     (zip [0 .. length alphas] alphas)
308:               -- debugSplice $! "genRecursions.argss: " ++ pprint (foldl AppE r <$> argss)
309:               pure $ foldl AppE r <$> argss
310:         else pure []
311:     False -> pure []
312: 
313: canRecurse :: Splice Bool
314: canRecurse = not . Map.null <$> gets args_rec_ctx
315: 
316: matchesGoal :: Type -> Goal -> Bool
317: matchesGoal type_ goal = maybe True (`compareTypes` type_) goal
