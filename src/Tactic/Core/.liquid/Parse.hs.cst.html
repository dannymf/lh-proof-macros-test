 26: type Parser a = P.ParsecT String () Q a
 27: 
 28: runParser :: Parser a -> String -> Q a
 29: runParser p str =
 30:   P.runParserT p () "SourceName" str >>= \case
 31:     Left err -> error $ show err
 32:     Right a -> pure a

 34: parseInstrs :: Parser [Instr]
 35: parseInstrs = do
 36:   instrs <- concat <$> parseInstr `P.sepBy` parseSymbol ";"
 37:   P.optionMaybe . P.try $ parseSymbol ";" -- because why not
 38:   pure instrs
 39: 
 40: -- intros of the form `[x1 x2 ... xn | ... | y1 y2 ... ym ]`
 41: parseIntros :: Parser [[String]]
 42: parseIntros = do
 43:   parseSymbol "["
 44:   -- s <- parseUntilBefore (parseIsChar ']')
 45:   -- let intros = fmap (splitByChar ' ') . splitByChar '|' $ s
 46:   intros <- (parseNameString `P.sepBy` P.spaces) `P.sepBy` (lexeme (P.char '/'))
 47:   parseSymbol "]"
 48:   pure intros
 49: 
 50: -- `[x1, x2, ...]:`
 51: parseRequires :: Parser (Maybe [String])
 52: parseRequires = P.optionMaybe . P.try $ do
 53:   parseSymbol "["
 54:   requires <- parseNameString `P.sepBy` parseSymbol ","
 55:   parseSymbol "]:"
 56: 
 57:   pure requires
 58: 
 59: parseFlags :: Parser [String]
 60: parseFlags = do
 61:   mb_start <- P.optionMaybe $ P.try $ parseSymbol "#["
 62:   case mb_start of
 63:     Just _ -> do
 64:       flags <- parseWord `P.sepBy` parseSymbol ","
 65:       parseSymbol "]"
 66:       pure flags
 67:     Nothing ->
 68:       pure []
 69: 
 70: parseInstrRequires :: [String] -> Parser [Instr]
 71: parseInstrRequires requires =
 72:   P.choice . fmap P.try $
 73:     [ -- Assert
 74:       do
 75:         parseSymbol "assert"
 76:         exp <- lexeme parseExp
 77:         pure [Assert {exp, requires}],
 78:       -- Dismiss
 79:       do
 80:         parseSymbol "dismiss"
 81:         exp <- lexeme parseExp
 82:         pure [Dismiss {exp, requires}],
 83:       -- Use
 84:       do
 85:         parseSymbol "use"
 86:         exp <- lexeme parseExp
 87:         pure [Use {exp, requires}],
 88:       -- Cond
 89:       do
 90:         parseSymbol "condition"
 91:         exp <- lexeme parseExp
 92:         pure [Cond {exp, requires}],
 93:       -- error
 94:       do
 95:         error "only these tactics allow a scope guard: assert, dismiss, use, condition"
 96:     ]

 98: parseInstr :: Parser [Instr]
 99: parseInstr = do
100:   parseRequires >>= \case
101:     Just requires -> parseInstrRequires requires
102:     Nothing -> parseInstr'
103: 
104: parseInstr' :: Parser [Instr]
105: parseInstr' =
106:   P.choice . fmap P.try $
107:     [ -- Intro
108:       do
109:         parseSymbol "intro"
110:         name <- parseNameString
111:         pure [Intro {name}],
112:       -- Destruct
113:       do
114:         parseSymbol "destruct"
115:         name <- parseNameString
116:         intros <- P.optionMaybe . P.try $ do
117:           parseSymbol "as"
118:           parseIntros
119:         flags <- parseFlags
120:         pure [Destruct {name, intros, flags}],
121:       -- Induct
122:       do
123:         parseSymbol "induct"
124:         name <- parseNameString
125:         intros <- P.optionMaybe . P.try $ do
126:           parseSymbol "as"
127:           parseIntros
128:         flags <- parseFlags
129:         pure [Induct {name, intros, flags}],
130:       -- Auto
131:       do
132:         parseSymbol "auto"
133:         mb_hints <- P.optionMaybe . P.try $ do
134:           parseSymbol "["
135:           hints <- fmap nameToExp <$> P.sepBy parseName (parseSymbol ",")
136:           parseSymbol "]"
137:           pure hints
138:         mb_depth <- P.optionMaybe . P.try $ parseInt
139:         pure [Auto {hints = maybe defaultAutoHints id mb_hints, depth = maybe defaultAutoDepth id mb_depth}],
140:       -- Assert
141:       do
142:         parseSymbol "assert"
143:         exp <- lexeme parseExp
144:         pure [Assert {exp, requires = []}],
145:       -- Dismiss
146:       do
147:         parseSymbol "dismiss"
148:         exp <- lexeme parseExp
149:         pure [Dismiss {exp, requires = []}],
150:       -- Use
151:       do
152:         parseSymbol "use"
153:         exp <- lexeme parseExp
154:         pure [Use {exp, requires = []}],
155:       -- Cond
156:       do
157:         parseSymbol "condition"
158:         exp <- lexeme parseExp
159:         pure [Cond {exp, requires = []}],
160:       -- Trivial
161:       do
162:         parseSymbol "trivial"
163:         pure [Trivial],
164:       -- comment
165:       do
166:         P.spaces
167:         P.string "--"
168:         P.manyTill P.anyChar (P.try P.newline)
169:         pure []
170:     ]

172: parseDecInstrs :: Parser (Environment, [Instr])
173: parseDecInstrs = do
174:   -- sig
175:   P.spaces
176:   def_name <- parseName
177:   P.string "::"
178:   P.many $ P.char ' '
179:   def_type_string <- parseUntilBefore parseNextIsNewline
180:   def_type <- fromMP $ MP.parseType def_type_string
181:   let (def_argTypes, _) = flattenType def_type
182:   -- imp
183:   _ <- parseName -- == def_name
184:   def_argNames <- P.many parseName
185:   parseSymbol "="
186:   instrs <- parseInstrs
187:   pure
188:     ( emptyEnvironment
189:         { def_name = def_name,
190:           def_type = def_type,
191:           def_argNames = def_argNames,
192:           def_argTypes = def_argTypes
193:         },
194:       ((\name -> Intro {name = show name}) <$> def_argNames)
195:         ++ instrs
196:     )

198: fromMP :: Either String a -> Parser a
199: fromMP e = case e of
200:   Right a -> pure a
201:   Left msg -> error $ "metaparse: " ++ msg
202: 
203: lexeme :: Parser a -> Parser a
204: lexeme p = do
205:   a <- p
206:   P.spaces
207:   pure a

209: parseSymbol :: String -> Parser String
210: parseSymbol = lexeme . P.string

212: -- characters allowed in ids
213: parseNameChar :: Parser Char
214: parseNameChar = P.alphaNum P.<|> P.oneOf "_'"
215: 
216: parseName :: Parser Name
217: parseName = lexeme do
218:   s <- P.many1 parseNameChar
219:   pure $ mkName s

221: parseNameString :: Parser String
222: parseNameString = lexeme do
223:   P.many1 parseNameChar
224: 
225: parseWord :: Parser String
226: parseWord = lexeme do
227:   P.many1 P.alphaNum
228: 
229: parseInt :: Parser Int
230: parseInt = lexeme do
231:   ds <- P.many1 P.digit
232:   pure $ read ds

234: parseIsChar :: Char -> Parser Bool
235: parseIsChar c = do
236:   c' <- P.anyChar
237:   pure $ c == c'

239: parseExp :: Parser Exp
240: parseExp = do
241:   rest <- lookAheadRest
242:   debugM $ "parseExp: " ++ rest
243:   -- str <- P.between (parseSymbol "{") (parseSymbol "}") (P.many1 P.anyChar)
244:   parseSymbol "{"
245:   str <- P.manyTill P.anyChar (P.try (parseSymbol "}"))
246:   debugM $ "str: " ++ str
247:   fromMP (MP.parseExp str)

249: parseNextIsNewline :: Parser Bool
250: parseNextIsNewline = do
251:   c <- P.lookAhead P.anyChar
252:   pure $ c == '\n'

254: -- parses the string until right before `p` parses `True`
255: parseUntilBefore :: Parser Bool -> Parser String
256: parseUntilBefore p = lexeme go
257:   where
258:     go = do
259:       b <- P.lookAhead p
260:       if b
261:         then pure ""
262:         else do
263:           c <- P.anyChar
264:           (c :) <$> go
265: 
266: lookAheadRest :: Parser String
267: lookAheadRest = P.lookAhead (P.many P.anyChar)
268: 
269: splitByChar :: Char -> String -> [String]
270: splitByChar sep str = go "" str
271:   where
272:     go "" "" = []
273:     go acc "" = [acc]
274:     go acc (c : str)
275:       | c == sep = acc : go "" str
276:       | otherwise = go (acc ++ [c]) str
