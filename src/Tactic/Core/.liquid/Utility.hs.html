10: flattenType :: Type -> ([Type], Type)
11: flattenType (AppT (AppT ArrowT alpha) beta) =
12:   let (alphas, delta) = flattenType beta
13:    in (alpha : alphas, delta)

25: -- because normal equality treats the types `ConT N` and `ConT Path.To.Module.N` as different, even if they are the same... how to fix this???
26: compareTypes :: Type -> Type -> Bool
27: compareTypes (ConT n1) (ConT n2) = nameBase n1 == nameBase n2
28: compareTypes (AppT a1 b1) (AppT a2 b2) = compareTypes a1 a2 && compareTypes b1 b2
29: compareTypes t1 t2 = t1 == t2

31: index :: [a] -> Int -> Maybe a
32: index [] _ = Nothing
33: index (x : xs) 0 = Just x
34: index (_ : xs) i = index xs (i - 1)

36: -- | for each list of the output, the ith element is from the ith list of the input.
37: -- all lists must be of the same length.
38: -- example: [ [x1, x2], [y1, y2] ] ==> [ [x1, y1], [x1, y2], [x2, y1], [x2, y2] ]
39: fanout :: [[a]] -> [[a]]
40: fanout [] = []
41: fanout (xs : []) = [[a] | a <- xs]
42: fanout (xs : xss) = [a' : xs' | a' <- xs, xs' <- fanout xss]

44: -- useMany [e1, e2, e3] == [|use e1 &&&& use e2 &&& use e3|]
45: useMany :: [Exp] -> Q Exp
46: useMany [] = [|trivial|]
47: useMany [e] = [|use $(pure e)|]
48: useMany (e : es) = [|use $(pure e) &&& $(useMany es)|]
49: 
50: conjunction :: [Exp] -> Q Exp
51: conjunction [] = [|trivial|]
52: conjunction [e] = pure e
53: conjunction (e : es) = [|$(pure e) &&& $(conjunction es)|]
54: 
55: conjunctionExp :: [Exp] -> Exp
56: conjunctionExp [] = VarE (mkName "trivial")
57: conjunctionExp [e] = e
58: conjunctionExp (e : es) = InfixE (Just e) (VarE (mkName "&&&")) (Just (conjunctionExp es))
